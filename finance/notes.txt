CREATE TABLE users (id INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL, username TEXT NOT NULL, hash TEXT NOT NULL, cash NUMERIC NOT NULL DEFAULT 10000.00);
CREATE TABLE sqlite_sequence(name,seq);
CREATE UNIQUE INDEX username ON users (username);
CREATE TABLE trades (id INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL, username TEXT NOT NULL, key TEXT NOT NULL, shares NUMERIC NOT NULL, price NUMERIC NOT NULL, time TEXT NOT NULL, FOREIGN KEY (username) REFERENCES users(id));
CREATE TABLE owned(id INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL, user_id INTEGER, key TEXT NOT NULL, shares NUMERIC NOT NULL, FOREIGN KEY (user_id) REFERENCES users(id));
